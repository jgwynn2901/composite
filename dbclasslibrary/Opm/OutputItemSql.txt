( 
	SELECT Field.NAME AS NAME, Item.SEQUENCE AS SEQUENCE, 
			Field.FONT_NAME AS FONT_NAME, Field.FONT_SIZE AS FONT_SIZE, 
			Field.BOLD_FLG AS BOLD_FLG, Field.ITALIC_FLG AS ITALIC_FLG, 
			Field.UNDERLINE_FLG AS UNDERLINE_FLG, Field.STRIKEOUT_FLG AS STRIKEOUT_FLG, 
			Field.WIDTH AS WIDTH, Field.HEIGHT AS HEIGHT, 
			Field.XPOS AS XPOS, Field.YPOS AS YPOS, 
			Field.MULTILINE_FLG AS MULTILINE_FLG, Page.PAGE_NUMBER AS PAGE_NUMBER, 
			Field.BMP AS BMP, Def.DUPLEX_PRINT_FLG AS DUPLEX_PRINT_FLG, 
			Page.OUTPUT_TRAY AS OUTPUT_TRAY, Page.BACKGROUND_BMP AS BACKGROUND_BMP, Page.ORIENTATION AS ORIENTATION, Page.NAME AS PAGE_NAME, 
			Field.MAPPING AS MAPPING, Item.RULE_ID AS RULE_ID 
	FROM 
		Output_Item Item, 
		Output_Field Field, 
		Output_Definition Def, 
		Output_Page Page 
	WHERE 
		Item.OUTPUTDEF_ID=Def.OUTPUTDEF_ID 
			AND Def.OUTPUTDEF_ID = Page.OUTPUTDEF_ID 
			AND Field.OUTPUT_PAGE_ID = Page.OUTPUT_PAGE_ID 
			AND Item.TRANSMISSION_SEQ_STEP_ID = :TranSeqId
			AND Field.OUTPUT_FLD_ID NOT IN 
			( 
    		SELECT distinct Output_Fld_ID 
    		FROM Output_Mapping Mapping, Routing_Plan Routing, Transmission_Seq_Step SeqStep 
    		WHERE Routing.ACCNT_HRCY_STEP_ID = Mapping.ACCNT_HRCY_STEP_ID 
					AND SeqStep.TRANSMISSION_SEQ_STEP_ID = :TranSeqId
					AND SeqStep.ROUTING_PLAN_ID = Routing.ROUTING_PLAN_ID
			)
UNION 
    SELECT Mapping.NAME AS NAME, Item.SEQUENCE AS SEQUENCE, 
			Mapping.FONT_NAME AS FONT_NAME, Mapping.FONT_SIZE AS FONT_SIZE, 
			Mapping.BOLD_FLG AS BOLD_FLG, Mapping.ITALIC_FLG AS ITALIC_FLG, 
			Mapping.UNDERLINE_FLG AS UNDERLINE_FLG, Mapping.STRIKEOUT_FLG AS STRIKEOUT_FLG, 
			Mapping.WIDTH AS WIDTH, Mapping.HEIGHT AS HEIGHT, 
			Mapping.XPOS AS XPOS, Mapping.YPOS AS YPOS, 
			Mapping.MULTILINE_FLG AS MULTILINE_FLG, Page.PAGE_NUMBER AS PAGE_NUMBER, 
			Mapping.BMP AS BMP, Def.DUPLEX_PRINT_FLG AS DUPLEX_PRINT_FLG, 
			Page.OUTPUT_TRAY AS OUTPUT_TRAY, Page.BACKGROUND_BMP AS BACKGROUND_BMP, Page.ORIENTATION AS ORIENTATION, Page.NAME AS PAGE_NAME, 
			Mapping.MAPPING AS MAPPING, Item.RULE_ID AS RULE_ID 
		FROM   
			Output_Item 
			Item, 
			Output_Field 
			Field, 
			Output_Definition Def, 
			Output_Page Page, 
			Output_Mapping Mapping, 
			Routing_Plan Routing, 
			Transmission_Seq_Step SeqStep 
	WHERE 
		Item.OUTPUTDEF_ID=Def.OUTPUTDEF_ID 
			AND Def.OUTPUTDEF_ID = Page.OUTPUTDEF_ID 
			AND Field.OUTPUT_PAGE_ID = Page.OUTPUT_PAGE_ID 
			AND Item.TRANSMISSION_SEQ_STEP_ID = :TranSeqId
			AND Field.OUTPUT_FLD_ID = Mapping.OUTPUT_FLD_ID 
			AND Routing.ACCNT_HRCY_STEP_ID = Mapping.ACCNT_HRCY_STEP_ID 
			AND SeqStep.ROUTING_PLAN_ID = Routing.ROUTING_PLAN_ID 
			AND SeqStep.TRANSMISSION_SEQ_STEP_ID = :TranSeqId	
) 
ORDER BY SEQUENCE, PAGE_NUMBER, HEIGHT
